	ruler_designer_cost = -666

	name = {
		first_valid = {
			triggered_desc = {
				trigger = { NOT = { exists = this } }
				desc = trait_p_dv_akatosh_base
			}
			desc = trait_p_dv_akatosh
		}
	}

	desc = {
		first_valid = {

			}
			desc = trait_p_dv_akatosh_character_desc
		}
	}

	icon = {
		first_valid = {
			triggered_desc = {
				trigger = { NOT = { exists = this } }
				desc = {
					first_valid = {
						triggered_desc = {
							trigger = { is_player_selected = yes }
							desc = "[Select_CString( GetPlayer.MakeScope.Var('faith_window').IsSet, Concatenate( 'patron_nbg/', GetPlayer.Custom('get_dv_akatosh_icon') ), 'p_dv_akatosh.dds' )]"

						desc = p_dv_akatosh.dds
