KnightCulture = {
	type = character
	
	text = {
		trigger = {
			OR = {
				AND = {
					has_government = holy_order_government
					culture = { has_cultural_pillar = language_dunmeris }
				}
				AND = {
					has_government = theocracy_government
					culture = { has_cultural_pillar = language_dunmeris }
				}
			}
				
		}
		localization_key = knight_ordinator
	}	

	text = {
		 trigger = {
			OR = {
				AND = {
					culture = { has_cultural_pillar = heritage_atmoran }
					culture = { has_cultural_pillar = language_nordic }
				}
				AND = {
					culture = { has_cultural_pillar = heritage_atmoran }
					culture = { has_cultural_pillar = language_cyrodiilic }
				}
			}
			 	NOT = { has_government = pirate_government } 
		 }
		 localization_key = knight_nord
	}
	
	text = {
		 trigger = {
			culture = { has_cultural_pillar = language_dunmeris }
				NOT = { has_government = pirate_government } 
		 }
		 localization_key = knight_dunmer
	}

	text = {
		trigger = {
			OR = {
				culture = { has_cultural_pillar = language_bretic }
				culture = { has_cultural_pillar = language_cyrodiilic }
				culture = { has_cultural_pillar = language_falmeris }
				faith = { religion_tag = trinimac_religion }
			}
				NOT = { has_government = pirate_government } 
		}
		localization_key = knight_default
	}
	
	text = {
		trigger = {
			OR = {
				has_government = pirate_government
			} 
		}
		localization_key = knight_pirate
	}

	}
